## VPS/Locally deploy!
>     $ apt-get -y update

>     $ apt-get -y install git gcc python3-pip -y

>     $ git clone https://github.com/Imshivaexe/RaBBiTXBot

>     $ pip3 install -U -r requirements.txt

>     $ cp sample.env .env

>     $ nano .env
# Edit .env with your own values and Run Bot
>     $ screen

>     $ python3 -m Imshivaexe
